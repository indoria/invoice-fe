2. IIFE Module Revealing JavaScript

This script defines a module GoalTracker using the IIFE and revealing module patterns. It includes a nested module for local storage handling. It provides methods to initialize the tracker with the JSON data, get the current tracking progress, and update milestones or metrics.


<script>
const GoalTracker = (function() {

    let strategyData = null;
    let trackingData = {};
    const STORAGE_KEY = 'invoicingGoalTrackerData';

    const LocalStorageManager = (function() {

        const save = function(data) {
            try {
                const serializedData = JSON.stringify(data);
                localStorage.setItem(STORAGE_KEY, serializedData);
            } catch (e) {
                console.error("Error saving data to local storage:", e);
            }
        };

        const load = function() {
            try {
                const serializedData = localStorage.getItem(STORAGE_KEY);
                if (serializedData === null) {
                    return null;
                }
                const data = JSON.parse(serializedData);
                return data;
            } catch (e) {
                console.error("Error loading data from local storage:", e);
                return null;
            }
        };

        return {
            save: save,
            load: load
        };

    })();

    const initializeTrackingData = function(data) {
         const initialData = {
            milestones: {},
            metrics: {}
         };

         if (data && data.plan && Array.isArray(data.plan)) {
             data.plan.forEach(phase => {
                 if (phase.milestones && Array.isArray(phase.milestones)) {
                     phase.milestones.forEach(milestone => {
                         initialData.milestones[milestone.name] = {
                             completed: false,
                             completionDate: null
                         };
                     });
                 }
             });
         }

         if (data && data.metrics && Array.isArray(data.metrics)) {
             data.metrics.forEach(metric => {
                 initialData.metrics[metric.name] = {
                     target: metric.targetValue,
                     currentValue: 0,
                     history: []
                 };
                 if (metric.name === "Monthly Recurring Revenue (MRR)" || metric.name === "Total Paying Customers") {
                     initialData.metrics[metric.name].currentValue = 0;
                 }
             });
         }
         return initialData;
    };

    const initialize = function(jsonData) {
        if (!jsonData) {
            console.error("Goal Tracker initialization failed: No strategy JSON provided.");
            return;
        }
        strategyData = jsonData;

        const loadedData = LocalStorageManager.load();

        if (loadedData) {
            trackingData = loadedData;
            console.log("Goal Tracker initialized. Tracking data loaded from local storage.");
        } else {
            trackingData = initializeTrackingData(strategyData);
            LocalStorageManager.save(trackingData);
            console.log("Goal Tracker initialized. New tracking data created.");
        }
    };

    const getStrategy = function() {
        return strategyData;
    };

    const getTrackingProgress = function() {
         return JSON.parse(JSON.stringify(trackingData));
    };

    const updateMilestoneStatus = function(milestoneName, completed) {
        if (trackingData.milestones && trackingData.milestones[milestoneName]) {
            trackingData.milestones[milestoneName].completed = completed;
            trackingData.milestones[milestoneName].completionDate = completed ? new Date().toISOString() : null;
            LocalStorageManager.save(trackingData);
            console.log(`Milestone "${milestoneName}" status updated to ${completed}.`);
            return true;
        }
        console.warn(`Milestone "${milestoneName}" not found in tracking data.`);
        return false;
    };

     const updateMetricValue = function(metricName, newValue) {
         if (trackingData.metrics && trackingData.metrics[metricName] !== undefined) {
            const metric = trackingData.metrics[metricName];
            const timestamp = new Date().toISOString();

            metric.history.push({ value: newValue, timestamp: timestamp });

            metric.currentValue = newValue;

            LocalStorageManager.save(trackingData);
            console.log(`Metric "${metricName}" value updated to ${newValue}. History recorded.`);
            return true;
        }
        console.warn(`Metric "${metricName}" not found in tracking data.`);
        return false;
     };

    return {
        init: initialize,
        getStrategy: getStrategy,
        getTrackingProgress: getTrackingProgress,
        updateMilestoneStatus: updateMilestoneStatus,
        updateMetricValue: updateMetricValue
    };

})();
</script>

How to Use:

  1. Save this content as an HTML file (e.g., goal_tracker.html).
  2. Open the HTML file in a web browser.
  3. Open your browser's developer console (usually F12).
  4. Observe the console logs. It will show the initialization, loading/saving from local storage, and updates.
  5. Refresh the page. The tracker should load the previously saved data from local storage.
  6. You can modify the myInvoicingStrategy JSON data structure or the example usage calls to see how the module responds.
  7. The data returned by GoalTracker.getTrackingProgress() can be used by other parts of your application (not included here) to visually render checklists, charts, etc.
  8. The nested LocalStorageManager handles persistence automatically whenever you call updateMilestoneStatus or updateMetricValue.
